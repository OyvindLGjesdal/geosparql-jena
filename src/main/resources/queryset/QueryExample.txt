WHERE {
?place ntu:hasExactGeometry ?geom .
?geom geo:asWKT ?WKT .

ntu:A ntu:hasExactGeometry ?aGeom .
?aGeom geo:asWKT ?aWKT .

FILTER ( geof:equals(?aWKT, ?WKT ) )
}
